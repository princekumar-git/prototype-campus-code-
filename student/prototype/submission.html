class Solution { public: vector<int> twoSum(vector<int>& nums, int target) { // Create a map to store the number and its index unordered_map<int, int> numMap; for (int i = 0; i < nums.size(); i++) { int complement = target - nums[i]; // If complement exists in map, we found the pair if (numMap.find(complement) != numMap.end()) { return {numMap[complement], i}; } // Store current number and index numMap[nums[i]] = i; } return {}; // Should not reach here } };
